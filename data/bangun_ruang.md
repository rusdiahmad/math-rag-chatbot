# Bangun Ruang - Dasar
## Kubus
Sisi = s
Volume = s^3
Luas permukaan = 6 * s^2

## Balok
Panjang = p, Lebar = l, Tinggi = t
Volume = p * l * t
Luas permukaan = 2*(p*l + p*t + l*t)

## Tabung
Jari-jari = r, Tinggi = t
Volume = π * r^2 * t
Luas permukaan = 2 * π * r * (r + t)

## Kerucut
Jari-jari = r, Tinggi = t, Garis pelukis = s
Volume = (1/3) * π * r^2 * t
Luas permukaan = π * r * (r + s)
